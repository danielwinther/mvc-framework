588  private function parseBlockScalar($style, $chomping = '', $indentation = 0)
589  {

599  while ($notEOF && $isCurrentLineBlank) {
600  // newline only if not EOF
601  if ($notEOF = $this->moveToNextLine()) {
602  $blockLines[] = '';
603  $isCurrentLineBlank = $this->isCurrentLineBlank();
604  }
605  }
606 
607  // determine indentation if not specified
608  if (0 === $indentation) {
609  if (preg_match('/^ +/', $this->currentLine, $matches)) {
610  $indentation = strlen($matches[0]);
611  }
612  }
613 
614  if ($indentation > 0) {
615  $pattern = sprintf('/^ {%d}(.*)$/', $indentation);
616 
617  while (
618  $notEOF && (
619  $isCurrentLineBlank ||
620  preg_match($pattern, $this->currentLine, $matches)
621  )
622  ) {
623  if ($isCurrentLineBlank && strlen($this->currentLine) > $indentation) {
624  $blockLines[] = substr($this->currentLine, $indentation);
625  } elseif ($isCurrentLineBlank) {
626  $blockLines[] = '';
627  } else {
628  $blockLines[] = $matches[1];
629  }
630 
631  // newline only if not EOF
632  if ($notEOF = $this->moveToNextLine()) {
633  $isCurrentLineBlank = $this->isCurrentLineBlank();
634  }
635  }
636  } elseif ($notEOF) {
637  $blockLines[] = '';
638  }
639 
640  if ($notEOF) {
641  $blockLines[] = '';
642  $this->moveToPreviousLine();
643  } elseif (!$notEOF && !$this->isCurrentLineLastLineInDocument()) {
644  $blockLines[] = '';
645  }
646 
647  // folded style
648  if ('>' === $style) {
649  $text = '';
650  $previousLineIndented = false;
651  $previousLineBlank = false;
652 
653  for ($i = 0; $i < count($blockLines); ++$i) {
654  if ('' === $blockLines[$i]) {
655  $text .= "\n";
656  $previousLineIndented = false;
657  $previousLineBlank = true;
658  } elseif (' ' === $blockLines[$i][0]) {
659  $text .= "\n".$blockLines[$i];
660  $previousLineIndented = true;
661  $previousLineBlank = false;
662  } elseif ($previousLineIndented) {
663  $text .= "\n".$blockLines[$i];
664  $previousLineIndented = false;
665  $previousLineBlank = false;
666  } elseif ($previousLineBlank || 0 === $i) {
667  $text .= $blockLines[$i];
668  $previousLineIndented = false;
669  $previousLineBlank = false;
670  } else {
671  $text .= ' '.$blockLines[$i];
672  $previousLineIndented = false;
673  $previousLineBlank = false;
674  }
675  }
676  } else {

678  }
679

681  if ('' === $chomping) {
682  $text = preg_replace('/\n+$/', "\n", $text);
683  } elseif ('-' === $chomping) {
684  $text = preg_replace('/\n+$/', '', $text);
